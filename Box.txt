	}
 	


	Box(int width, int length, int height) {

	}
	
	Box(String color, int width, int length, int height) {
		//super(); //기본값으로 없으면 자동으로 컴파일러가 삽입해줌
		this.color = color;
		this.width = width;
		this.length = length;
		this.height = height;
	}

		return color;
	}

		this.color = color;
	}

		System.out.println("상자를 만든 이:" + this.maker+", 색깔, 가로, 세로, 높이는 " 
	                      + ", " + color + width + ", " +

	}

	@Override //상속받아 그대로 사용안하고 재정의
	public String toString() {
		return "박스 [color=" + color + ", width=" + width + ", length=" + length + ", height=" + height + "]";
	}

}
